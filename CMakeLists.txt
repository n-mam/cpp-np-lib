project (NPL)

find_package(Threads)

if (UNIX)
  find_package(OpenSSL REQUIRED)
endif (UNIX)

ADD_EXECUTABLE(ftp ftp.cpp)

SET_PROPERTY(TARGET ftp PROPERTY CXX_STANDARD 17)

TARGET_INCLUDE_DIRECTORIES(ftp PRIVATE INCLUDE)

TARGET_INCLUDE_DIRECTORIES(ftp PUBLIC .)

if (WIN32)
  TARGET_LINK_LIBRARIES(ftp Ws2_32.lib)
else()
  TARGET_LINK_LIBRARIES(ftp ${CMAKE_THREAD_LIBS_INIT} OpenSSL::SSL)
endif (WIN32)
